# Use Python 3.11 slim image
FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Set environment variables
ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    PYTHONPATH=/app

# --- 1. System Dependencies (CRITICAL FIX: Changed from Postgres to MySQL) ---
# Install system dependencies (gcc, default-libmysqlclient-dev) needed for MySQL drivers
RUN apt-get update && apt-get install -y \
    gcc \
    default-libmysqlclient-dev \
    pkg-config \
    # Clean up APT lists
    && rm -rf /var/lib/apt/lists/*

# --- 2. Copy Files (Matching your run.py and requirements.txt) ---
COPY requirements.txt .
COPY run.py .
# If your application logic requires the 'app' directory, uncomment the line below:
# COPY app /app/app

# --- 3. Python Dependencies ---
# This installs all packages, including the necessary MySQL driver (e.g., aiomysql, if in requirements.txt)
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# --- 4. Security Setup ---
RUN adduser --disabled-password --gecos '' appuser && \
    chown -R appuser:appuser /app
USER appuser

# Expose port
EXPOSE 8000

# Health check
HEALTHCHECK --interval=30s --timeout=30s --start-period=5s --retries=3 \
    CMD ["curl", "-f", "http://localhost:8000/health"] || exit 1

# --- 5. Application Entry Point ---
# Runs the application, pointing Uvicorn to run.py
CMD ["uvicorn", "run:app", "--host", "0.0.0.0", "--port", "8000"]
